% ФАКТЫ
собака(флэш).
собака(ровер).
кошка(бутси).
лошадь(стар).

цвет(флэш, черный).
цвет(бутси, коричневый).
цвет(ровер, рыжий).
цвет(стар, белый).

% ПРАВИЛА
домашнее_животное(X) :- собака(X).
домашнее_животное(X) :- кошка(X).

животное(X) :- домашнее_животное(X).
животное(X) :- лошадь(X).

% У Тома есть собака не черного цвета.
владелец(том, Y) :- собака(Y), not(цвет(Y, черный)).

% У Кейта есть лошадь или что-то черного цвета.
владелец(кейт, Z) :- лошадь(Z).
владелец(кейт, Z) :- цвет(Z, черный).
