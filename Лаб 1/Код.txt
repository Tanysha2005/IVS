% ЗАДАНИЕ 1 

% Факты о животных
dog('Флэш').
dog('Ровер').
cat('Бутси').
horse('Стар').

% Факты о цвете животных
color('Флэш', черный).
color('Бутси', коричневый).
color('Ровер', рыжий).
color('Стар', белый).

% Домашнее животное - это собака или кошка
pet(Животное) :- dog(Животное).
pet(Животное) :- cat(Животное).

% Животное - домашнее животное или лошадь
animal(Животное) :- pet(Животное).
animal(Животное) :- horse(Животное).

% У Тома есть собака не черного цвета
owns('Том', Питомец) :- dog(Питомец), color(Питомец, Цвет), Цвет \= черный.

% У Кейта есть лошадь или что-то черного цвета
owns('Кейт', Питомец) :- horse(Питомец).
owns('Кейт', Питомец) :- color(Питомец, черный).
